export function RingSchematic() {
  const cell = 18;
  const cells = [];
  for (let r = 0; r < 16; r++) {
    for (let c = 0; c < 16; c++) {
      cells.push({ x: c, y: r, frame: r === 0 || r === 15 || c === 0 || c === 15 });
    }
  }
  return (
    <div className="mx-auto max-w-xs">
      <svg viewBox={`0 0 ${16 * cell} ${16 * cell}`} className="w-full rounded-lg border bg-card">
        {cells.map((c, i) => (
          <rect
            key={i}
            x={c.x * cell + 1}
            y={c.y * cell + 1}
            width={cell - 2}
            height={cell - 2}
            rx={2}
            className={c.frame ? "fill-amber-400" : "fill-muted"}
          />
        ))}
      </svg>
    </div>
  );
}

export function RareSwatches() {
  return (
    <div className="flex flex-wrap justify-center gap-4">
      {data.rareColors.map((c) => (
        <div key={c} className="flex flex-col items-center gap-1">
          <MotifSwatch color={c} width={52} />
          <span className="font-mono text-xs text-muted-foreground">'{colorToLetter(c)}'</span>
        </div>
      ))}
    </div>
  );
}

export function FrameInteriorChart() {
  const isClient = useIsClient();
  const chartData = data.colors.map((c) => ({ color: c.color, frame: c.frame, interior: c.interior }));
  return (

              dataKey="color"

              fontSize={11}
            />

              label={{ value: "arêtes", angle: -90, position: "insideLeft", fontSize: 12 }}

Les 22 couleurs ne jouent pas à armes égales. Classez-les selon l'endroit où
vivent leurs arêtes : deux familles se dessinent aussitôt. Cinq couleurs
rares, assignées à résidence sur le bord ; dix-sept couleurs communes, qui
font presque tout le travail au cœur du plateau.

## Pourquoi c'est important

Cette structure a été voulue, et elle se voit. Eternity I est tombé parce que
ses pièces présentaient des régularités statistiques exploitables ; pour
Eternity II, les couleurs ont été équilibrées à dessein, précisément pour
supprimer ces prises. Cantonner les couleurs rares au bord, c'est les tenir à
l'écart du vaste intérieur, là où un attaquant rêverait justement d'un
signal rare et très contraignant à exploiter. L'intérieur est abandonné aux
dix-sept couleurs communes : voilà pourquoi la recherche y trouve si peu de
prise. Et le bord, où les couleurs rares resserrent les accords, explique
pourquoi tout plateau de haut niveau affiche un bord quasi parfait.
